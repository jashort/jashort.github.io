Joining paths with a leading slash overwrite the path they're being joined with

Joining a directory with a leading slash doesn’t necessarily do what you’d think.

import os

os.path.join("/tmp/", "other")
# '/tmp/other'
os.path.join("/tmp/", "other")
# '/tmp/other'
os.path.join("/tmp/", "/other")
# '/other'  (?!)
import pathlib

tmp = pathlib.Path("/tmp")
other = pathlib.Path("other")
tmp / other
# PosixPath('/tmp/other')
other = pathlib.Path("/other")
tmp / other
# PosixPath('/other')
